bool PacRenderer::ImportPacAiExport(const std::string& exportFolderPath) {
    namespace fs = std::filesystem;

    if (!fs::exists(exportFolderPath)) {
        std::cerr << "[PacRenderer] Export folder not found: " << exportFolderPath << std::endl;
        return false;
    }

    // 1. Load visual manifest
    std::string manifestPath = exportFolderPath + "/visual_manifest.json";
    VisualManifest manifest;

    if (!VisualManifestLoader::Load(manifestPath, manifest)) {
        std::cerr << "[PacRenderer] Failed to load visual_manifest.json\n";
        return false;
    }

    // 2. Load core PacData (simulation)
    std::string pacdataPath = exportFolderPath + "/world.pacdata.json";
    // TODO: Call your existing PacData loader here
    // PacDataWorld world = PacDataLoader::Load(pacdataPath);

    // 3. Setup environment (sky, fog, lighting)
    m_scene->SetEnvironment(manifest.environment);

    // 4. Load static meshes
    for (const auto& meshData : manifest.static_meshes) {
        std::string fullPath = exportFolderPath + "/" + meshData.asset;
        auto* proxy = m_scene->CreateProxy(0); // Use proper entity ID later
        if (proxy) {
            // Load glTF mesh + material
            if (GltfLoader::LoadMesh(fullPath, proxy)) {
                // Apply transform
                // proxy->transform = ... from meshData.transform
                proxy->castShadows = true;
            }
        }
    }

    // 5. Load entities with render components
    for (const auto& entityData : manifest.entities) {
        auto* proxy = m_scene->CreateProxy(entityData.id);
        if (proxy && !entityData.render.asset.empty()) {
            std::string fullAssetPath = exportFolderPath + "/" + entityData.render.asset;
            
            if (GltfLoader::LoadMesh(fullAssetPath, proxy)) {
                // Apply material overrides from manifest
                ApplyMaterialOverrides(proxy, entityData.render.material_overrides);
            }
        }
    }

    std::cout << "[PacRenderer] Successfully imported PacAi export from: " 
              << exportFolderPath << std::endl;
    std::cout << "   Entities: " << manifest.entities.size() 
              << " | Static Meshes: " << manifest.static_meshes.size() << std::endl;

    return true;
}
