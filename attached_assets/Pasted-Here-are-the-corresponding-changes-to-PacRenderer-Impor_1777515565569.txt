bool PacRenderer::ImportPacAiExport(const std::string& exportFolderPath) {
    if (!fs::exists(exportFolderPath)) {
        std::cerr << "[PacRenderer] Export folder not found: " << exportFolderPath << std::endl;
        return false;
    }

    std::cout << "[PacRenderer] Importing PacAi export from: " << exportFolderPath << std::endl;

    // 1. Ensure assets folder exists
    fs::path assetsDir = fs::path(exportFolderPath) / "assets";
    if (!fs::exists(assetsDir)) {
        fs::create_directories(assetsDir);
        std::cout << "[PacRenderer] Created assets directory: " << assetsDir << std::endl;
    }

    // 2. Load Visual Manifest (for rendering)
    std::string manifestPath = exportFolderPath + "/visual_manifest.json";
    VisualManifest manifest;

    if (!VisualManifestLoader::Load(manifestPath, manifest)) {
        std::cerr << "[PacRenderer] Failed to load visual_manifest.json" << std::endl;
        // Continue anyway - we can still load simulation data
    } else {
        // Apply environment settings
        m_scene->SetEnvironment(manifest.environment);
    }

    // 3. Create placeholder assets if they don't exist (for immediate visual feedback)
    CreatePlaceholderAssets(assetsDir);

    // 4. Load core PacData (simulation world)
    std::string pacdataPath = exportFolderPath + "/world.pacdata.json";
    // TODO: Integrate with your existing PacData loader
    // bool success = LoadPacData(pacdataPath);
    std::cout << "[PacRenderer] Loading PacData: " << pacdataPath << std::endl;

    // 5. Load visual entities and static meshes from manifest
    LoadVisualEntities(manifest);
    LoadStaticMeshes(manifest, exportFolderPath);

    std::cout << "[PacRenderer] Import completed successfully." << std::endl;
    return true;
}

C++// Helper: Create placeholder glTF assets so the scene isn't empty
void PacRenderer::CreatePlaceholderAssets(const fs::path& assetsDir) {
    // Create basic agent model placeholder
    fs::path agentPath = assetsDir / "models" / "agent.gltf";
    fs::create_directories(agentPath.parent_path());
    
    if (!fs::exists(agentPath)) {
        // For now, just create an empty file as placeholder
        // In real workflow, you would generate or copy real glTF files here
        std::ofstream placeholder(agentPath);
        placeholder << "// Placeholder glTF for agent\n";
        placeholder.close();
        std::cout << "[PacRenderer] Created placeholder: " << agentPath << std::endl;
    }

    // Create ground/arena placeholder
    fs::path groundPath = assetsDir / "models" / "terrain" / "arena.gltf";
    fs::create_directories(groundPath.parent_path());
    
    if (!fs::exists(groundPath)) {
        std::ofstream placeholder(groundPath);
        placeholder << "// Placeholder glTF for arena terrain\n";
        placeholder.close();
        std::cout << "[PacRenderer] Created placeholder: " << groundPath << std::endl;
    }
}

// Load entities defined in visual_manifest.json
void PacRenderer::LoadVisualEntities(const VisualManifest& manifest) {
    for (const auto& [entityId, renderComp] : manifest.entities) {
        if (renderComp.asset.empty()) continue;

        RenderProxy* proxy = m_scene->CreateProxy(entityId);
        if (proxy) {
            std::string fullPath = renderComp.asset; // relative path from export root
            if (GltfLoader::LoadMesh(fullPath, proxy)) {
                // Apply material overrides if any
                // TODO: Implement material override logic
                proxy->visible = renderComp.visible;
                proxy->castShadows = renderComp.cast_shadows;
            }
        }
    }
}
